                Try

                    ' Insertar tabla clientes_sii
                    Dim query As String = "INSERT INTO clientes_SII (RUT_cliente, cliente, email_XML) VALUES (@RUT_cliente, @cliente, @email_XML)"
                    Using conexion As New SqlConnection(conexString)
                        Using comando As New SqlCommand(query, conexion)
                            ' Parámetros para evitar SQL Injection
                            comando.Parameters.AddWithValue("@RUT_cliente", campos(0))
                            comando.Parameters.AddWithValue("@cliente", campos(1))

                            comando.ExecuteNonQuery()
                        End Using
                    End Using
                Catch ex As Exception
                    Continue While
                End Try
            End While
        End Using
